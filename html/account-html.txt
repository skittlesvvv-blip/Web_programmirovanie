    <!-- ==================== ЛОГИКА АККАУНТА + ИСТОРИЯ ЗАКАЗОВ ==================== -->
    <script>
    document.addEventListener('DOMContentLoaded', function() {
        const user = Auth?.getCurrentUser?.() || Storage.get(Storage.keys.CURRENT_USER);
        
        if (!user) {
            window.location.href = 'login.html';
            return;
        }

        // Заполнение данных пользователя
        document.getElementById('accountGreeting').textContent = `Добро пожаловать, ${user.name || ''}!`;
        document.getElementById('userName').textContent = user.name || '—';
        document.getElementById('userSurname').textContent = user.surname || '—';
        document.getElementById('userEmail').textContent = user.email || '—';
        document.getElementById('userPhone').textContent = user.phone || '—';

        if (user.createdAt) {
            document.getElementById('userRegistered').textContent = new Date(user.createdAt).toLocaleDateString('ru-RU');
        }

        // ====================== ИСТОРИЯ ЗАКАЗОВ ======================
        function renderOrders() {
            const ordersList = document.getElementById('ordersList');
            const noOrders = document.getElementById('noOrders');
            const userKey = user.email || user.username || user.id || String(user);

            const allOrders = Storage.get(Storage.keys.ORDERS) || {};
            const userOrders = allOrders[userKey] || [];

            if (userOrders.length === 0) {
                noOrders.style.display = 'block';
                ordersList.style.display = 'none';
                return;
            }

            noOrders.style.display = 'none';
            ordersList.style.display = 'block';

            let html = '';
            userOrders.forEach(order => {
                html += `
                <div class="order-item">
                    <div class="order-header">
                        <span class="order-id">Заказ № ${order.id}</span>
                        <span class="order-date">${order.date}</span>
                    </div>
                    <div class="order-total">Сумма: <strong>${order.total} BYN</strong></div>
                    <div class="order-status">Статус: <span class="status-new">Новый</span></div>
                </div>`;
            });

            ordersList.innerHTML = html;
        }

        renderOrders();

        // Лояльность, колесо и т.д. (твой код)
        if (typeof Loyalty !== 'undefined') Loyalty.renderProgress();

        // Выход
        document.getElementById('logoutBtn').onclick = () => {
            if (confirm('Вы действительно хотите выйти?')) {
                Auth.logout?.();
            }
        };

        console.log('✅ Аккаунт загружен с историей заказов');
    });
    </script>
