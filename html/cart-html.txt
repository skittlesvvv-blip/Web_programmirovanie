<script>
document.addEventListener('DOMContentLoaded', function() {
    const btn = document.getElementById('btnCheckout');
    const note = document.getElementById('checkoutNote');

    function updateCheckoutButton() {
        if (!btn || !note) return;

        const user = Auth?.getCurrentUser?.() || Storage.get(Storage.keys.CURRENT_USER);
        const userKey = user?.email || user?.username || user || 'unknown';

        if (userKey !== 'unknown') {
            btn.disabled = false;
            btn.style.opacity = '1';
            btn.style.cursor = 'pointer';
            note.style.display = 'none';

            btn.onclick = function() {
                console.log('%c=== ОФОРМЛЕНИЕ ЗАКАЗА ===', 'color: orange; font-weight: bold');
                console.log('Текущий пользователь:', userKey);

                // 1. Пробуем через Cart модуль
                let cart = [];
                if (typeof Cart !== 'undefined' && typeof Cart.getCart === 'function') {
                    cart = Cart.getCart() || [];
                }

                // 2. Если пусто — берём из Storage
                if (cart.length === 0) {
                    const cartData = Storage.get(Storage.keys.CART) || {};
                    console.log('Storage.CART содержит:', cartData);

                    // Пробуем по точному ключу пользователя
                    if (Array.isArray(cartData[userKey])) {
                        cart = cartData[userKey];
                    } 
                    // Если не нашли — ищем любой непустой массив (на случай разных ключей)
                    else {
                        for (let key in cartData) {
                            if (Array.isArray(cartData[key]) && cartData[key].length > 0) {
                                cart = cartData[key];
                                console.log('Найдена корзина по ключу:', key);
                                break;
                            }
                        }
                    }
                }

                console.log('Итоговая корзина:', cart);
                console.log('Количество товаров:', cart.length);

                if (cart.length === 0) {
                    alert('Ваша корзина пуста');
                    return;
                }

                // Создаём заказ
                const total = cart.reduce((sum, item) => sum + (Number(item.price) || 0) * (item.quantity || 1), 0);

                const order = {
                    id: 'ORDER-' + Date.now(),
                    date: new Date().toLocaleDateString('ru-RU'),
                    items: cart,
                    total: total,
                    status: 'new'
                };

                // Сохраняем заказ
                let orders = Storage.get(Storage.keys.ORDERS) || {};
                if (!orders[userKey]) orders[userKey] = [];
                orders[userKey].unshift(order);
                Storage.set(Storage.keys.ORDERS, orders);

                // Очищаем корзину
                const cartData = Storage.get(Storage.keys.CART) || {};
                cartData[userKey] = [];
                Storage.set(Storage.keys.CART, cartData);

                if (typeof Cart !== 'undefined' && typeof Cart.clear === 'function') Cart.clear();

                alert('✅ Заказ успешно оформлен!\n\nНомер заказа: ' + order.id);
                window.location.href = 'account.html';
            };
        } else {
            btn.disabled = true;
            btn.style.opacity = '0.5';
            btn.style.cursor = 'not-allowed';
            note.style.display = 'block';
        }
    }

    updateCheckoutButton();

    window.addEventListener('storage', (e) => {
        if (e.key === Storage.keys.CURRENT_USER) updateCheckoutButton();
    });
});
</script>
