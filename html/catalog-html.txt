<script>
    // === Ф-9 Поиск + Фильтрация (All / Classic / Printed) ===
    function filterProducts() {
        const searchText = document.getElementById('searchInput').value.toLowerCase().trim();
        const activeCategory = document.querySelector('.filter-btn.active').getAttribute('data-category');

        document.querySelectorAll('.product-showcase, .products-grid__item').forEach(item => {
            const name = (item.getAttribute('data-name') || '').toLowerCase();
            const category = item.getAttribute('data-category') || '';

            let matchesSearch = true;
            if (searchText) {
                const words = searchText.split(/\s+/).filter(w => w.length > 0);
                matchesSearch = words.every(word => name.includes(word));
            }

            let matchesCategory = true;
            if (activeCategory !== 'all') {
                matchesCategory = category === activeCategory;
            }

            item.style.display = (matchesSearch && matchesCategory) ? '' : 'none';
        });
    }

    function clearSearch() {
        document.getElementById('searchInput').value = '';
        filterProducts();
    }

    document.addEventListener('DOMContentLoaded', () => {
        const searchInput = document.getElementById('searchInput');

        searchInput.addEventListener('input', () => {
            document.querySelector('.clear-btn').style.display = searchInput.value ? 'block' : 'none';
            filterProducts();
        });

        document.querySelectorAll('.filter-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                filterProducts();
            });
        });
    });
</script>
