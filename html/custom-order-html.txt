<script>
    document.addEventListener('DOMContentLoaded', function() {
        const form = document.getElementById('customOrderFormEl');
        if (!form) return;
        
        // Валидация в реальном времени
        const inputs = form.querySelectorAll('input[required], textarea[required]');
        inputs.forEach(input => {
            input.addEventListener('blur', validateField);
            input.addEventListener('input', () => {
                const errorEl = document.getElementById(input.id.replace('custom', '') + 'Error');
                if (errorEl) errorEl.classList.remove('show');
                input.classList.remove('input-error');
            });
        });
        
        // Обработчик отправки формы
        form.addEventListener('submit', async function(e) {
            e.preventDefault();
            
            // Валидация всех полей
            let isValid = true;
            inputs.forEach(input => {
                if (!validateField(input)) isValid = false;
            });
            
            // Проверка согласия
            const agree = document.getElementById('customAgree');
            const agreeError = document.getElementById('agreeError');
            if (!agree.checked) {
                agreeError.textContent = window.i18n?.t?.('error_agree') || 'Необходимо согласие';
                agreeError.classList.add('show');
                isValid = false;
            }
            
            if (!isValid) {
                showNotification?.(window.i18n?.t?.('error_form') || 'Пожалуйста, исправьте ошибки в форме');
                return;
            }
            
            // Сбор данных
            const formData = {
                id: Date.now(),
                name: document.getElementById('customName').value.trim(),
                email: document.getElementById('customEmail').value.trim(),
                phone: document.getElementById('customPhone').value.trim(),
                description: document.getElementById('customDescription').value.trim(),
                sketch: document.getElementById('customSketch').files[0]?.name || null,
                status: 'new',
                date: new Date().toISOString().split('T')[0],
                userId: Auth?.getCurrentUser?.()?.id || null
            };
            
            try {
                // Сохранение в JSON Server (или localStorage для демо)
                if (typeof Storage !== 'undefined') {
                    const orders = Storage.get('scala_customOrders') || [];
                    orders.push(formData);
                    Storage.set('scala_customOrders', orders);
                }
                
                // Показать сообщение об успехе
                document.getElementById('successMessage').style.display = 'block';
                form.reset();
                form.scrollIntoView({ behavior: 'smooth' });
                
                showNotification?.(window.i18n?.t?.('custom_success') || 'Заявка отправлена!');
                
            } catch (err) {
                console.error('Ошибка отправки заявки:', err);
                showNotification?.(window.i18n?.t?.('error_submit') || 'Ошибка отправки. Попробуйте позже.');
            }
        });
        
        // Функция валидации поля
        function validateField(input) {
            const value = input.value.trim();
            const errorEl = document.getElementById(input.id.replace('custom', '') + 'Error');
            let isValid = true;
            
            if (!value && input.required) {
                errorEl.textContent = window.i18n?.t?.('error_required') || 'Обязательное поле';
                isValid = false;
            } else if (input.pattern && value && !new RegExp(input.pattern).test(value)) {
                if (input.type === 'email') {
                    errorEl.textContent = window.i18n?.t?.('error_email') || 'Некорректный email';
                } else if (input.type === 'tel') {
                    errorEl.textContent = window.i18n?.t?.('error_phone') || 'Формат: +375(29)1234567';
                } else {
                    errorEl.textContent = window.i18n?.t?.('error_format') || 'Неверный формат';
                }
                isValid = false;
            }
            
            if (!isValid) {
                errorEl.classList.add('show');
                input.classList.add('input-error');
            } else {
                errorEl.classList.remove('show');
                input.classList.remove('input-error');
            }
            
            return isValid;
        }
    });
