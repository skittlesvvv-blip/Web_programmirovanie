<script>
document.addEventListener('DOMContentLoaded', function() {
    const form = document.getElementById('registerForm');
    if (!form) return;

    let nicknameAttempts = 0;
    const MAX_NICKNAME_ATTEMPTS = 5;

    function generateNickname() {
        const adjectives = ['wild', 'bold', 'free', 'swift', 'grace', 'luxe', 'pure', 'vivid'];
        const nouns = ['soul', 'spirit', 'vibe', 'essence', 'aura', 'glow', 'pulse', 'flow'];
        const num = Math.floor(Math.random() * 999);
        return adjectives[Math.floor(Math.random() * adjectives.length)] + 
               nouns[Math.floor(Math.random() * nouns.length)] + num;
    }

    function setNickname(nick) {
        document.getElementById('nickname').value = nick;
    }

    setNickname(generateNickname());

    document.getElementById('btnRegenerate').addEventListener('click', function() {
        if (nicknameAttempts < MAX_NICKNAME_ATTEMPTS) {
            nicknameAttempts++;
            setNickname(generateNickname());
            if (nicknameAttempts === MAX_NICKNAME_ATTEMPTS) {
                document.getElementById('nickname').removeAttribute('readonly');
                document.getElementById('nickname').placeholder = 'Введите свой никнейм';
                this.textContent = '✏️ Изменить';
            }
        } else {
            document.getElementById('nickname').focus();
        }
    });

    const inputs = form.querySelectorAll('input[required]');
    inputs.forEach(input => {
        input.addEventListener('blur', () => validateField(input));
        input.addEventListener('input', () => {
            clearError(input.id);
            checkFormValidity();
        });
    });

    document.getElementById('passwordConfirm').addEventListener('blur', function() {
        const pwd = document.getElementById('password').value;
        const confirm = this.value;
        const errorEl = document.getElementById('passwordConfirmError');

        if (confirm && pwd !== confirm) {
            errorEl.textContent = 'Пароли не совпадают';
            errorEl.classList.add('show');
            this.classList.add('input-error');
        } else {
            errorEl.classList.remove('show');
            this.classList.remove('input-error');
        }
        checkFormValidity();
    });

    form.addEventListener('submit', async function(e) {
        e.preventDefault();

        document.getElementById('formError').style.display = 'none';
        document.getElementById('successMessage').style.display = 'none';

        let isValid = true;
        inputs.forEach(input => {
            if (!validateField(input)) isValid = false;
        });

        const agree = document.getElementById('agree');
        if (!agree.checked) {
            document.getElementById('agreeError').textContent = 'Необходимо согласие';
            document.getElementById('agreeError').classList.add('show');
            isValid = false;
        }

        if (!isValid) {
            if (typeof showNotification === 'function') {
                showNotification('Пожалуйста, исправьте ошибки в форме', 'error');
            }
            return;
        }

        const userData = {
            id: Date.now(),
            email: document.getElementById('email').value.trim(),
            password: document.getElementById('password').value,
            name: document.getElementById('name').value.trim(),
            surname: document.getElementById('surname').value.trim(),
            phone: document.getElementById('phone').value.trim(),
            birthDate: document.getElementById('birthDate').value.trim(),
            nickname: document.getElementById('nickname').value.trim() || generateNickname(),
            subscribe: document.getElementById('subscribe').checked,
            role: 'buyer',
            createdAt: new Date().toISOString()
        };

        try {
            const users = Storage.get(Storage.keys.USERS) || [];

            if (users.find(u => u.email === userData.email)) {
                if (typeof showFormError === 'function') showFormError('Пользователь с таким email уже существует');
                return;
            }

            users.push(userData);
            Storage.set(Storage.keys.USERS, users);
            Storage.set(Storage.keys.CURRENT_USER, { ...userData, password: undefined });

            document.getElementById('successMessage').textContent = 'Регистрация успешна!';
            document.getElementById('successMessage').style.display = 'block';

            setTimeout(() => {
                location.href = 'index.html';
            }, 1200);

        } catch (err) {
            console.error(err);
            if (typeof showFormError === 'function') showFormError('Ошибка регистрации. Попробуйте позже.');
        }
    });

    function validateField(input) {
        const value = input.value.trim();
        const errorEl = document.getElementById(input.id + 'Error');
        let isValid = true;

        if (!value && input.required) {
            errorEl.textContent = 'Обязательное поле';
            isValid = false;
        } else if (input.id === 'password') {
            if (value.length < 6) {
                errorEl.textContent = 'Минимум 6 символов';
                isValid = false;
            } else if (!/\d/.test(value)) {
                errorEl.textContent = 'Должен содержать хотя бы одну цифру';
                isValid = false;
            }
        } else if (input.pattern && value && !new RegExp(input.pattern).test(value)) {
            if (input.id === 'email') errorEl.textContent = 'Некорректный email';
            else if (input.id === 'phone') errorEl.textContent = 'Формат: +375(29)1234567';
            else errorEl.textContent = 'Неверный формат';
            isValid = false;
        }

        if (!isValid) {
            errorEl.classList.add('show');
            input.classList.add('input-error');
        } else {
            errorEl.classList.remove('show');
            input.classList.remove('input-error');
        }

        return isValid;
    }

    function clearError(fieldId) {
        const errorEl = document.getElementById(fieldId + 'Error');
        const inputEl = document.getElementById(fieldId);
        if (errorEl) errorEl.classList.remove('show');
        if (inputEl) inputEl.classList.remove('input-error');
    }

    function checkFormValidity() {
        const required = ['email', 'password', 'passwordConfirm', 'name', 'surname', 'birthDate'];
        const allFilled = required.every(id => document.getElementById(id).value.trim());
        const match = document.getElementById('password').value === document.getElementById('passwordConfirm').value;
        const agree = document.getElementById('agree').checked;

        document.getElementById('btnRegister').disabled = !(allFilled && match && agree);
    }
});
