const Reviews = {
    KEYS: {
        REVIEWS: 'scala_reviews'
    },

    getAllReviews(productId) {
        const all = Storage.get(this.KEYS.REVIEWS) || {};
        return all[productId] || [];
    },

    saveReviews(productId, reviews) {
        let all = Storage.get(this.KEYS.REVIEWS) || {};
        all[productId] = reviews;
        Storage.set(this.KEYS.REVIEWS, all);
    },

    async addReview(productId, productName, rating, text) {
        const user = Auth?.getCurrentUser?.() || Storage.get(Storage.keys.CURRENT_USER);
        if (!user) {
            showNotification?.(I18n.t('product_login_required'), 'error');
            return { success: false };
        }

        const reviews = this.getAllReviews(productId);

        const newReview = {
            id: Date.now().toString(36),
            productId,
            productName,
            userName: user.name || 'Клиент',
            rating: parseInt(rating),
            text: text.trim(),
            date: new Date().toISOString()
        };

        reviews.unshift(newReview);
        this.saveReviews(productId, reviews);

        showNotification?.(I18n.t('review_success'), 'success');
        return { success: true };
    },

    getAverageRating(productId) {
        const reviews = this.getAllReviews(productId);
        if (reviews.length === 0) return 0;
        const sum = reviews.reduce((acc, r) => acc + r.rating, 0);
        return (sum / reviews.length).toFixed(1);
    },

    render(productId, containerId = 'reviews-list') {
        const container = document.getElementById(containerId);
        if (!container) return;

        const reviews = this.getAllReviews(productId);
        const avg = this.getAverageRating(productId);

        let html = `<div class="reviews-header">
            <h3 data-i18n="reviews_title">Отзывы</h3>
            ${avg > 0 ? `<div class="avg-rating">⭐ ${avg}</div>` : ''}
        </div>`;

        if (reviews.length === 0) {
            html += `<p class="no-reviews" data-i18n="reviews_empty">Пока нет отзывов. Будьте первым!</p>`;
        } else {
            html += reviews.map(r => `
                <div class="review-card">
                    <div class="review-top">
                        <span class="review-author">${r.userName}</span>
                        <span class="review-date">${new Date(r.date).toLocaleDateString('ru-RU')}</span>
                    </div>
                    <div class="review-stars">${'★'.repeat(r.rating)}${'☆'.repeat(5 - r.rating)}</div>
                    <p class="review-text">${r.text}</p>
                </div>
            `).join('');
        }

        container.innerHTML = html;

        // ← Добавлено: применяем перевод i18n после рендера
        if (typeof I18n !== 'undefined') I18n.setLanguage(I18n.currentLang);
    }

// Глобальная функция
window.submitReview = function(productId, productName) {
    const rating = document.getElementById('reviewRating').value;
    const text = document.getElementById('reviewText').value.trim();

    if (!rating || !text) {
        showNotification?.(I18n.t('review_form_error'), 'error');
        return;
    }

    Reviews.addReview(productId, productName, rating, text).then(() => {
        document.getElementById('reviewText').value = '';
        Reviews.render(productId);
    });
};
